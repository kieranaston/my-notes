    XML (local)

    import xml.etree.ElementTree as ET

def parse_xml(path):
    tree = ET.parse(p)
    root = tree.getroot()
    return pd.DataFrame([{el.tag: el.text for el in row} for row in root])

    df_xml = parse_xml('file_name.xml')
